\section{Outils}

\subsection{Dérivée Totale}

$$\frac{\DD\bullet}{\DD t} = \frac{\dr\bullet}{\dr t} + \vv{u}\vv{\nabla}\bullet$$

\subsection{Impédance Ramenée}

$$Z = Z_c\frac{Z_L + jZ_C\tan kL}{Z_C + jZ_L\tan kL}$$

\subsection{Conservation de la Masse}

$$\frac{\dr\rho}{\dr t} + \vv{\nabla}(\rho\vv{u}) = 0$$

\subsection{Équation du Navier-Stokes}

$$\rho\frac{\DD\vv{u}}{\DD t} = - \vv{\nabla}p + \vv{\nabla}\uul{\tau} + \vv{f}$$
$$\uul{\tau} = \eta(\uul{\nabla}\vv{u} + (\uul{\nabla}\vv{u})^T) - \frac{2}{3}\eta(\uul{\nabla}\vv{u})\uul{I}$$

En négligeant les pertes visqueuses (\textbf{Équation d'Euler}):

$$p\frac{\DD\vv{u}}{\DD t} = -\vv{\nabla}p$$

\subsection{Équation d'Euler (forme de Crocco)}

$$\frac{\dr\vv{u}}{\dr t} = -\vv{\nabla}B - \vv{\nabla}\wedge\vv{\nabla}\wedge\vv{u}$$
$$B = \frac{\dr\rho}{\rho} + \frac{u^2}{2}$$

\subsection{Conservation de l'Énergie}

$$\rho T\frac{\DD S}{\DD t} = - \vv{\nabla}\cdot\vv{q} - \uul{\tau}:\uul{u}$$

\begin{itemize}
	\item $q$ : flux de chaleur
	\item $S$ : entropie
\end{itemize}

\subsection{Équation d'État}

$$\frac{\DD P}{\DD t} = \left.\frac{\dr P}{\dr\rho}\right)_S\frac{\DD\rho}{\DD t} + \left.\frac{\dr P}{\dr S}\right)_\rho\frac{\DD S}{\DD t}$$

\section{Équation d'Onde}

Écoulement uniforme $\Rightarrow$ équation d'onde convectée

$$\frac{1}{c_0^2}\frac{\DD^2p}{\DD t^2} - \Delta p = 0$$

\section{Solutions}

\subsection{Sol. Modales avec écoulement}

$$p = \sum_{n=0}^\infty c_n\psi_n(y,z)e^{-jk_0z}$$
$$\Delta_\bot\psi_n = -\alpha^2\psi_nn  ~~,~~ \alpha_n^2 = \left(\frac{\omega}{c_0} - Mk_n\right) - k_n^2$$

\subsection{Nombre d'onde (Guide 2D)}

\subsubsection{Sans écoulement}

$$k_n = \sqrt{\left(\frac{\omega}{c_0}\right)^2 - \left(\frac{n\pi}{h}\right)^2}$$

\subsubsection{Écoulement uniforme}

$$k_n = \frac{-\omega M \pm \sqrt{\omega^2 - (1-M^2)\alpha_n^2}}{1-M^2}$$

\subsection{Nombre d'onde (Rect. 3D)}

$$l_{mn}^2 = (\frac{\omega}{c_0} - Mk_{mn})^2 - \left(\frac{n\pi}{a}\right) - \left(\frac{m\pi}{a\beta}\right)$$

\begin{itemize}
	\item $\beta$ aspect ratio
\end{itemize}
