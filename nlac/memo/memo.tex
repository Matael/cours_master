\begin{center}
     \Large{\underline{Non Linear Acoustics}} \\
\end{center}

\section{Governing Equations}

\subsubsection{Mass Conservation}

$$\frac{\dr\rho}{\dr t} + \div(\rho\vv{v}) = 0$$

\subsubsection{Momentum Conservation}

$$\rho\frac{\DD\vv{v}}{\DD t} = -\grad p + \eta\vv{\Delta}\vv{v} = \left(\xi +\frac{1}{3}\eta\right)\grad\div\vv{v}$$

\subsection{Energy Conservation}

$$\rho T\left(\frac{\dr S}{\dr t} + (\vv{v}\vv{\nabla})S\right) = \lambda\Delta T + \xi(\div\vv{v})^2 +
\frac{\eta}{2}\left(\frac{\dr v_i}{\dr x_k} + \frac{\dr v_k}{\dr x_i} - \frac{2}{3}\delta_{ik}\frac{\dr v_l}{\dr
x_l}\right)$$

\subsection{State Equation}

$$P = P(\rho,S)$$

\section{Simple Wave Equation}

$$\frac{\dr v}{\dr x} - \frac{\epsilon}{c_0^2}v\frac{\dr v}{\dr \tau} = 0
~~\Rightarrow~~
\frac{\dr V}{\dr \Xi} - V\frac{\dr V}{\dr \Theta} = 0$$

$$\epsilon = \frac{\gamma + 1}{2} ~~,~~ \tau = x-\frac{t}{c_0}$$
$$\Theta	 = \tau\omega_0 ~~,~~ V = \frac{v}{v_0} ~~,~~ \Xi = \frac{x}{x_{NL}}$$

\section{Simple Wave Equation with absorption}

$$\zeta,\eta\not=0 ~,~ \lambda\not=0 ~~\Rightarrow~~ \frac{\dr^2v}{\dr t^2} - c_0^2\frac{\dr^2v}{\dr x^2} =
\frac{b}{c_0} \frac{dr^3v}{\dr x^2\dr t}$$

\subsection{Effective Viscosity}

$$b = \zeta + \frac{4}{3}\eta +\lambda\left(\frac{1}{C_V}-\frac{1}{C_P}\right)$$

\section{NL Effects \& Absorption -- Burger's Equation}

$$\frac{\dr v}{\dr x} - \frac{\epsilon}{c_0^2}v\frac{\dr v}{\dr t} - \frac{b}{2\rho_0c_0^3}\frac{\dr^2v}{\dr t^2} = 0$$
$$\frac{\dr V}{\dr\Xi} - V\frac{\dr V}{\dr\Theta} - \underbrace{\frac{1}{2\epsilon\mathrm{Re}}}_G\frac{\dr^2V}{\dr\Theta^2} = 0$$

\subsection{About $\mathrm{Re}$ -- Acoustical Reynolds}

$$\mathrm{Re} = \frac{\rho_0c_0v_0}{b\omega_0}$$

\begin{itemize}
	\item $\mathrm{Re} << 1$ Viscosity hides NL effects ($V\nicefrac{\dr V}{\dr\Theta}$ is small)
	\item $\mathrm{Re} >> 1$ indicates $\frac{1}{2\epsilon\mathrm{Re}}\nicefrac{\dr^2V}{\dr\Theta^2}$ is neglectable for
		short distances
\end{itemize}
\section{Kordeveg de Vries-Burger's Equation}

$$\frac{\dr V}{\dr\Xi} - V\frac{\dr V}{\dr\Theta} - G\frac{\dr^2V}{\dr\Theta^2} + \underbrace{D\frac{\dr^3V}{\dr\Theta^3}}_{dissipation} = 0$$
